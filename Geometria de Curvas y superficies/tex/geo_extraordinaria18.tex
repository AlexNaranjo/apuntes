\begin{questions}
  \question Sea $\alpha(t)$ una curva en $\mathbb{R}^3$ con rapidez unitaria,
  triedro de Frenet-Serret $\left\{\textbf{t}(t), \textbf{n}(t), \textbf{b}(t)
  \right\}$, curvatura $\kappa(t) > 0$ y torsión $\tau(t) \neq 0$ para todo
  $t$. Suponemos que la curvatura y la torsión de $\alpha(t)$ verifcican la
  ecuación

  \begin{equation*}
    \dfrac{\tau(t)}{\kappa(t)} + \left(\dfrac{1}{\tau(t)} \left(
    \dfrac{1}{\kappa(t)}\right)'\right)' = 0
  \end{equation*}
  Demuestra:
  \begin{parts}
    \part $c(t) = \alpha(t) + \dfrac{1}{\kappa(t)}\textbf{n}(t) -
    \dfrac{\kappa'(t)}{\tau(t)(\kappa(t))^2}\textbf{b}(t) \quad$ es constante.

    \part $\alpha(t)$ pertenece a una esfera. Determina el centro y el radio
    de la esfera. (Demuestra que no dependen de $t$)
  \end{parts}

  \vfill

  \question Considera el helicoide $S$ parametrizada por $\textbf{X}(u,v) =
  (u\cos v, u\sin v, v)$

  \begin{parts}
    \part Calcula la primera y segunda formas fundamentales.

    \part Determina el coseno del ángulo entre las curvas
    \begin{equation*}
      \alpha(t) = \textbf{X}\left(t, t+1\right) \quad \text{y} \quad \beta(x)
      = \textbf{X}\left(\dfrac{x^2}{2}-1, x\right)
    \end{equation*}
    en el punto de intersección \textbf{X}(1,2).

    \part Determina la curvatura normal $k_n(t)$ de $\alpha(t) = \textbf{X}
    \left(t, t\right).$

    \part Encuentra las curvaturas principales y las direcciones principales
    de curvatura de la superficie $S$.

    \part Determina y resuelve la exuación diferencial que debe cumplir
    $\gamma(t) = \textbf{X}(u(t), v(t))$ para ser (i) línea de curvatura. (ii)
    línea asintótica.

    \textit{Recuerda:} $(arc\sinh x)' = \dfrac{1}{\sqrt{1+x^2}}$
  \end{parts}

  \vfill

  \question Considera el dominio $\Omega = \left\{(u,v) \in \mathbb{R}^2 :
  0<u<1\right\}$ con la métrica
  \begin{equation*}
    Q = \left(\dfrac{1}{u^2} - 1\right)\left((du)^2 + (dv)^2\right)
  \end{equation*}

  \begin{parts}
    \part Encuentra las geodésicas. (Exprésalas implícitamente, con $v$ en
    función de $u$).

    \textit{Sugerencia:} Utiliza el método de conservación del momento.

    \part Calcula la curvatura geodésica \textit{(escalar)} de la curva
    $\alpha(t) = (\frac{1}{2}, t^2)$ para $t>0$.
  \end{parts}

  \vfill

  \question Considera el semiplano superior $\mathbb{H} = \left\{(u,v) \in
  \mathbb{R}^2 : v > 0\right\}$ con la métrica $Q = \dfrac{(du)^2 +
  (dv)^2}{v^2}$.

  \begin{parts}
    \part Calcula la curvatura gaussiana de $\mathbb{H}$ usando formas
    diferenciales.

    \part Sea $h:\mathbb{H} \rightarrow \mathbb{H}$ la transformación definida
    por $h(u,v)=(\lambda(u, v), \nu(u, v))$ con
    \begin{equation*}
      \lambda(u,v) = \dfrac{u}{u^2+v^2} \quad \text{y} \quad \nu(u, v) =
      \dfrac{v}{u^2+v^2}
    \end{equation*}
    Estuia si $h$ es o no una isometría. \textit{Sugerencia:} Demuestra que
    $\lambda_u = -\nu_v$ y $\lambda_v = \nu_u$.
  \end{parts}

\end{questions}
