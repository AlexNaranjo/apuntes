\begin{tikzpicture}
\pgfmathsetmacro{\radius}{2}
\draw[] (-\radius,0) circle[radius=\radius];
\draw[] (\radius,0) circle[radius=\radius];

\foreach \c in {-\radius, \radius}
{
	\foreach \a in {0, 30, ..., 360}
		\draw[dense directed, orange] ({\c}, 0) -- ({\c + \radius * cos(\a)}, {\radius * sin(\a)});
}
\end{tikzpicture}
