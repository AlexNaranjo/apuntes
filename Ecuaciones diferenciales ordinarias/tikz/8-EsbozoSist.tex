{
	\foreach \y/\s/\c in {-2/<-/red,-1/<-/red,1/->/blue,2/->/blue}
	{
		\draw[blue, thick, \s, \c] ({\x}, {\y}) -- ({\x + 1}, {\y});
	}
}

	\draw[green!80!blue, thick] ({\x}, {0.2}) -- ({\x}, {-0.2});

\node[blue, anchor=west] at (2,3) {$y > 0 \implies$ soluciones crecientes en $x$};
